26  package net.sf.prefixedproperties;
27  
28  import java.io.IOException;
29  import java.text.MessageFormat;
30  import java.util.Locale;
31  import java.util.ResourceBundle;
32  
33  import junit.framework.Assert;
34  
35  import org.junit.Test;
36  
37  public class PrefixedPropertiesResourceBundleTest {
38  
39  	@Test
40  	public void testResourceBundle() throws IOException {
41  
42  		Locale.setDefault(Locale.TAIWAN); // we have to set another default
43  											// locale otherwise we will get a
44  											// german or english resourcebundle
45  											// as fallback.
46  		final ResourceBundle r1 = ResourceBundle.getBundle("rv", Locale.CHINESE);
47  		Assert.assertEquals("okay", r1.getString("test"));
48  
49  		final PrefixedResourceBundle rb3 = PrefixedResourceBundle.getPrefixedResourceBundle("rv", Locale.CHINESE,
50  				"local");
51  		Assert.assertEquals("ok", rb3.getString("test"));
52  
53  		final PrefixedResourceBundle rb = PrefixedResourceBundle.getPrefixedResourceBundle("rv", Locale.GERMANY,
54  				"local");
55  		Assert.assertEquals("wahr", rb.getString("test"));
56  
57  		final PrefixedResourceBundle rb2 = PrefixedResourceBundle.getPrefixedResourceBundle("rv", Locale.ENGLISH,
58  				"local");
59  		Assert.assertEquals("true", rb2.getString("test"));
60  
61  		final PrefixedResourceBundle rb4 = PrefixedResourceBundle.getPrefixedResourceBundle("rv", Locale.GERMANY,
62  				"local");
63  		Assert.assertEquals("wahr", rb4.getString("test"));
64  
65  		rb4.getPrefixedProperties().setLocalPrefix("male");
66  		Assert.assertEquals("Hallo Herr Marco", MessageFormat.format(rb4.getString("greetings"), "Marco"));
67  
68  		rb4.getPrefixedProperties().setLocalPrefix("female");
69  		Assert.assertEquals("Hallo Frau Marco", MessageFormat.format(rb4.getString("greetings"), "Marco"));
70  
71  		rb2.getPrefixedProperties().setLocalPrefix("male");
72  		Assert.assertEquals("Hello Mr. Marco", MessageFormat.format(rb2.getString("greetings"), "Marco"));
73  
74  		rb2.getPrefixedProperties().setLocalPrefix("female");
75  		Assert.assertEquals("Hello Mrs. Marco", MessageFormat.format(rb2.getString("greetings"), "Marco"));
76  	}
77  }
