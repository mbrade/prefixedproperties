229 		props2.setLocalPrefix("test");

234 			props2.setLocalPrefix("test");
235 			Assert.assertEquals("test", props2.getEffectivePrefix());
236 			props2.setLocalPrefix("test.something..wrong");
237 			Assert.fail();
238 		} catch (final Exception e) {
239 			Assert.assertEquals("test", props2.getEffectivePrefix());
240 		}
241 		props2.setLocalPrefix("test.anything.cache");
242 		Assert.assertEquals("test.anything.cache", props2.getEffectivePrefix());
243 		props2.setLocalPrefix("*");
244 		Assert.assertEquals("", props2.getEffectivePrefix());
245 
246 		props2.setLocalPrefix("test.anything.cache");
247 		Assert.assertEquals("test.anything.cache", props2.getEffectivePrefix());
248 		props2.setDefaultPrefix("dev.nothing.database");
249 		Assert.assertEquals("test.anything.cache", props2.getEffectivePrefix());
250 		props2.setLocalPrefix("*");// clears the configuredprefix
251 		Assert.assertEquals("dev.nothing.database", props2.getEffectivePrefix());// will
252 																					// fall
253 																					// to
254 																					// the
255 																					// defaults
256 		props2.setLocalPrefix("test.anything.cache");
257 		Assert.assertEquals("test.anything.cache", props2.getEffectivePrefix());
258 		props2.setDefaultPrefix("*"); // clear the defaults
259 		Assert.assertEquals("test.anything.cache", props2.getEffectivePrefix());
260 		props2.setLocalPrefix("*");// clears the configuredprefix
261 		Assert.assertEquals("", props2.getEffectivePrefix()); // since fallback
262 																// is cleared,
263 																// the
264 																// configuredprefix
265 																// will be
266 																// empty.

273 	@Test
274 	public void testCloneAndEquals() {
275 		properties.setProperty(getPrefixKey("KEYA"), "TEST-A");
276 		properties.setProperty("KEYA", "A");
277 		properties.setProperty(StagingPrefixConfig.LIVE + PrefixConfig.PREFIXDELIMITER + "C", "LIVE-C");
278 		properties.setProperty(getPrefixKey("KEYB"), "TEST-B");
279 		properties.put(new Integer(5), new Integer(50));
280 		Assert.assertTrue(properties.contains("TEST-A"));
281 		Assert.assertTrue(properties.contains("TEST-B"));
282 		Assert.assertFalse(properties.contains("KEYA"));
283 		Assert.assertFalse(properties.contains("TESTC"));
284 		Assert.assertTrue(properties.contains(new Integer(50)));
285 		Assert.assertFalse(properties.contains(new Integer(5)));
286 		Assert.assertEquals(properties.clone(), properties);
287 		Assert.assertEquals(properties.clone().hashCode(), properties.hashCode());
288 		final PrefixedProperties props2 = properties.clone();
289 		Assert.assertTrue(props2.contains("TEST-A"));
290 		Assert.assertTrue(props2.contains("TEST-B"));
291 		Assert.assertFalse(props2.contains("KEYA"));
292 		Assert.assertFalse(props2.contains("TESTC"));
293 		Assert.assertTrue(props2.contains(new Integer(50)));
294 		Assert.assertFalse(props2.contains(new Integer(5)));
295 		Assert.assertNotSame(props2, properties);
296 		props2.getPrefixConfig().setPrefix(StagingPrefixConfig.DEV);
297 		Assert.assertFalse(props2.equals(properties));
298 
299 	}

420 		}
