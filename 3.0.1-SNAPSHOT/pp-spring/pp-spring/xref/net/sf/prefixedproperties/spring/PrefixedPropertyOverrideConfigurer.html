48  /**
49   * The PrefixedPropertyOverrideConfigurer behaves the same as {@link PropertyOverrideConfigurer} with the difference that you can. To configure the default prefix you can use the following methods:
50   * {@link #setEnvironmentFactory(EnvironmentFactory)}<br/>
51   * {@link #setDefaultPrefixSystemPropertyKey(String)}<br/>
52   * {@link #setDefaultPrefix(String)}<br/>
53   * The usage of the methods will be in the same order as above.

87      /**
88       * Creates the properties.
89       * 
90       * @return the prefixed properties
91       */
92      protected synchronized PrefixedProperties createProperties() {
93  	if (myProperties == null) {
94  	    PrefixedProperties resultProperties = null;
95  	    String environment = defaultPrefix;
96  	    if (environmentFactory != null) {
97  		environment = environmentFactory.getEnvironment();
98  	    } else if (defaultPrefixSystemPropertyKey != null) {
99  		environment = System.getProperty(defaultPrefixSystemPropertyKey);
100 		if (environment == null) {
101 		    if (logger.isWarnEnabled()) {
102 			logger.warn(String.format("Didn't found system property key to set default prefix: %1s", defaultPrefixSystemPropertyKey));
103 		    }
104 		}
105 	    }
106 	    if (prefixConfigList != null) {
107 		resultProperties = PrefixedProperties.createCascadingPrefixProperties(prefixConfigList);
108 	    } else {
109 		if (environment != null) {
110 		    resultProperties = PrefixedProperties.createCascadingPrefixProperties(environment);
111 		} else {
112 		    resultProperties = new PrefixedProperties();
113 		}
114 	    }
115 	    resultProperties.setDefaultPrefix(environment);
116 	    if (logger.isInfoEnabled()) {
117 		logger.info(String.format("Setting default prefix to: %1s", environment));
118 	    }
119 	    resultProperties.setMixDefaultAndLocalPrefixSettings(mixDefaultAndLocalPrefixConfigurations);
120 	    myProperties = resultProperties;
121 	}
122 	return myProperties;
123     }

152     /**
153      * Load properties.
154      * 
155      * @param props
156      *            the props
157      * @throws IOException
158      *             Signals that an I/O exception has occurred.
159      */
160     @Override
161     protected void loadProperties(final Properties props) throws IOException {
162 	if (locations != null) {
163 	    for (int i = 0; i < locations.length; i++) {
164 		final Resource location = locations[i];
165 		if (logger.isInfoEnabled()) {
166 		    logger.info("Loading properties file from " + location);
167 		}
168 		File file = null;
169 		InputStream is = null;
170 		try {
171 		    try {
172 			file = location.getFile();
173 			is = new BufferedInputStream(new FileInputStream(file));
174 		    } catch (final IOException ie) {//ignore
175 		    } finally {
176 			if (file == null) {
177 			    is = location.getInputStream();
178 			}
179 		    }
180 
181 		    if (location.getFilename().toLowerCase().endsWith(Constants.XML_FILE_EXTENSION)) {
182 			persister.loadFromXml(props, is);
183 		    } else if (location.getFilename().toLowerCase().endsWith(Constants.JSON_FILE_EXTENSION)) {
184 			if (fileEncoding != null) {
185 			    persister.loadFromJson(props, new InputStreamReader(is, Charset.forName(fileEncoding)));
186 			} else {
187 			    persister.loadFromJson(props, is);
188 			}
189 		    } else {
190 			if (fileEncoding != null) {
191 			    persister.load(props, new InputStreamReader(is, Charset.forName(fileEncoding)));
192 			} else {
193 			    persister.load(props, is);
194 			}
195 		    }
196 		} catch (final IOException ex) {
197 		    if (ignoreResourceNotFound) {
198 			if (logger.isWarnEnabled()) {
199 			    logger.warn("Could not load properties from " + location + ": " + ex.getMessage());
200 			}
201 		    } else {
202 			throw ex;
203 		    }
204 		} finally {
205 		    if (is != null) {
206 			is.close();
207 		    }
208 		}
209 	    }
210 	}
211     }

213     /* (non-Javadoc)
214      * @see org.springframework.core.io.support.PropertiesLoaderSupport#mergeProperties()
215      */
216     @Override
217     protected Properties mergeProperties() throws IOException {
218 	final PrefixedProperties myProperties = createProperties();
219 	if (localOverride) {
220 	    loadProperties(myProperties);
221 	}
222 
223 	if (localProperties != null) {
224 	    for (int i = 0; i < localProperties.length; i++) {
225 		CollectionUtils.mergePropertiesIntoMap(localProperties[i], myProperties);
226 	    }
227 	}
228 
229 	if (!localOverride) {
230 	    loadProperties(myProperties);
231 	}
232 
233 	return myProperties;
234     }
